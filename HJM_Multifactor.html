import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import minimize

# Function to simulate HJM model
def hjm_multifactor(params, initial_curve, tenors, time_step, num_steps, num_paths):
    num_factors = len(params) // 2
    mean_reversion = params[:num_factors]
    volatility = params[num_factors:]
    
    
    # Generate correlated Brownian motions
    dW = np.random.normal(0, np.sqrt(time_step), (num_steps, num_factors, num_paths))
    
    # Initialize interest rates matrix
    rates = np.zeros((num_steps + 1, len(tenors), num_paths))
    rates[0, :, :] = initial_curve[:, np.newaxis]
    
    # Discretize the model using Euler method
    for t in range(num_steps):
        for i, tenor in enumerate(tenors):
            drift = -np.sum(mean_reversion * (rates[t, i, :] - 0.03) * time_step) 
            diffusion = np.dot(volatility, dW[t, :, :]) * np.sqrt(time_step)
            rates[t + 1, i, :] = rates[t, i, :] + drift + diffusion
    
    return rates

# Calibration function for HJM model
def hjm_multifactor_calibration(params, data, tenors, time_step, num_steps):
    initial_curve = np.atleast_1d(data.iloc[0])
    simulated_rates = hjm_multifactor(params, initial_curve, tenors, time_step, num_steps, num_paths=1)

    # Flatten the simulated rates and data for easy comparison
    simulated_rates_flat = simulated_rates[:, :, :].flatten()
    data_flat = data.values.flatten()

    diff = simulated_rates_flat[:len(data_flat)] - data_flat

    return np.sum(diff ** 2)  # Use sum of squared differences

# Load combined data (Interest rate data)
combined_interest_data = pd.read_csv('interest_rate_data.csv', index_col='Date', parse_dates=True) / 100
tenors = ['3-Month', '5-Year', '10-Year', '30-Year']

# Set time step for simulation
time_step_hjm_multi = 1

# Define calibration and forecasting periods
calibration_end_date_hjm = '2022-12-31'
forecasting_start_date = '2023-01-01'

# Extract data for calibration
calibration_data_hjm = combined_interest_data.loc[:calibration_end_date_hjm]

# Set number of steps for calibration
num_steps_calibration_hjm = len(calibration_data_hjm) - 1 #as we are using 2000 to 2022 for calibration

# Initial guess for parameters
initial_guess_hjm_multi = [0.0099] * len(tenors) + [0.051] * len(tenors) # list of mean reversion and volatility parameters

# Set a seed for reproducibility
seed = 50  # We can choose any integer

# Calibrate using historical data from 2000 to 2022 for each tenor
calibrated_params_hjm_multi = minimize(hjm_multifactor_calibration, initial_guess_hjm_multi,
                                       args=(calibration_data_hjm, tenors, time_step_hjm_multi, num_steps_calibration_hjm)).x


# Filter data for 2023
calibration_data_2023 = combined_interest_data.loc[forecasting_start_date:]


# Monte Carlo simulation for 2023
num_paths_hjm_multi = 5000
simulated_rates_paths = np.zeros((num_steps_calibration_hjm, len(tenors), num_paths_hjm_multi)) # Initialize an array to store simulated rates for each path

for i in range(num_paths_hjm_multi):
    initial_curve = calibration_data_2023.iloc[0].values
    simulated_rates = hjm_multifactor(calibrated_params_hjm_multi, initial_curve, tenors, time_step_hjm_multi, num_steps_calibration_hjm, 1)
    simulated_rates_paths[:, :, i] = simulated_rates[:num_steps_calibration_hjm, :, 0]

# Aggregate the results (e.g., mean or median)
forecasted_data_hjm_multi = pd.DataFrame({
    tenor: np.mean(simulated_rates_paths[:, j, :], axis=1)[:len(calibration_data_2023)] for j, tenor in enumerate(tenors)
}, index=calibration_data_2023.index)


# Calculate differences and metrics for 2023
differences_2023 = calibration_data_2023 - forecasted_data_hjm_multi
mean_difference_2023 = differences_2023.mean()
std_dev_difference_2023 = differences_2023.std()
rmse_per_tenor_2023 = np.sqrt(np.mean(differences_2023**2, axis=0))

# Visualize the results for each tenor in 2023
for tenor in tenors:
    plt.figure(figsize=(12, 6))
    plt.plot(calibration_data_2023.index, calibration_data_2023[tenor], label=f"{tenor} - Original")
    plt.plot(forecasted_data_hjm_multi.index, forecasted_data_hjm_multi[tenor], label=f"{tenor} - Forecasted")
    plt.title(f"HJM Multifactor Model Forecast for 2023 - {tenor} Tenor")
    plt.xlabel("Date")
    plt.ylabel("Interest Rate")
    plt.legend()
    plt.show()

print("Mean Reversion Parameters:", calibrated_params_hjm_multi[:len(tenors)])
print("Volatility Parameters:", calibrated_params_hjm_multi[len(tenors):])
